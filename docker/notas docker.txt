$Para correr docker compose entrando al bash
docker-compose up

$levantar servicio docker sin entrar a bash
docker-compose up -d

$para entrar a la consloa de docker
docker exec -it tramitescrd-web-1 bash

$Listar volumenes
docker volume ls 

$Eliminar contenendor
docker compose down -v

$Crear contenendor
docker compose up -d db
#up es para levantar, d es detach osea en segundo plano db es el nombre del servicio que se definio

$Crear contenendor APP
docker compose build tramitescrd
docker compose up tramitescrd


## Docker para postgres
- En ubuntu descargar docker compose
docker-compose run web django-admin startproject djangodockertest .
docker-compose run web


docker-compose up -d
docker exec -it tramitescrd-web-1 bash

$para levantar el contendedor con el servicio especifico
docker-compose up -d postgres

$para verificar que si esté corriendo el contendedor (inspeccionar que cosas están corriendo)
docker_compose ps

$para bajar un servicio específico
docker_compose down postgres

$Para que la BD postgres guarde información dentro de un contenedor se tien que crear un volumen

$Si se quiere verificar la ubicación de una imagen se hace en hub.docker.com/postgres

$El volumes es para que la informacion persista  quede guardada data en BD despues de bajar el servicio, 
para esto se tiene que cerar la carpeta donde se guardara la data y se tiene que ignorar si no se quiere
que la data se suba a git

$para conectarse a la BD
docker-compose postgres bash
psql -h localhost -d my_store -U nico
#donde h es host d base y U usuario

$para saber estructura de BD de postgres
\d+

$Para salirse de la BD
\q

$salirse de contenedor bash
exit
